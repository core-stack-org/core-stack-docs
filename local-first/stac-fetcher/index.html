    STAC Fetcher Reference¶

    The STAC (SpatioTemporal Asset Catalog) Fetcher downloads geospatial assets from the public CoRE Stack catalog for local processing.

    How It Works¶

    1. Query Building¶

    sequenceDiagram
    participant E as Engine
    participant SF as STAC Fetcher
    participant SC as STAC Catalog

    E->>SF: Request Data
    SF->>SF: Build Query
    SF->>SC: Search Items
    SC->>SF: Item Collection
    SF->>SF: Filter Assets
    SF->>E: Asset List
    from computing.stac_fetcher import STACFetcher

fetcher = STACFetcher()

# Build query
query = {
    'pipeline': 'swb',
    'state': 'karnataka',
    'district': 'raichur',
    'block': 'devadurga',
    'year': 2023
}

# Get required assets
assets = fetcher.get_pipeline_assets(query)
# Returns: ['lulc', 'mws', 'block_boundary']

    2. Cache Check¶

    graph TD
    A[Asset Request] --> B{In Cache?}
    B -->|Yes| C[Return Path]
    B -->|No| D[Download]
    D --> E[Save to Cache]
    E --> C

    style C fill:#e8f5e9
    def check_cache(self, asset_id: str) -> Optional[Path]:
    """Check if asset exists in local cache."""
    cache_path = self.cache_dir / f"{asset_id}.tif"

    if cache_path.exists():
        # Check freshness
        age_days = (time.time() - cache_path.stat().st_mtime) / 86400
        if age_days < self.cache_expiry_days:
            return cache_path

    return None

    3. Download¶

    def download_asset(self, asset_url: str, local_path: Path) -> Path:
    """Download asset with progress tracking."""

    response = requests.get(asset_url, stream=True)
    total_size = int(response.headers.get('content-length', 0))

    with open(local_path, 'wb') as f:
        downloaded = 0
        for chunk in response.iter_content(chunk_size=8192):
            f.write(chunk)
            downloaded += len(chunk)

            # Report progress
            progress = (downloaded / total_size) * 100
            self.report_progress(asset_url, progress)

    return local_path

    Error Handling¶

    Common Errors¶

    graph TD
    A[Fetch Error] --> B{Error Type}

    B -->|Network| C[Retry with Backoff]
    B -->|Not Found| D[Check Parameters]
    B -->|Corrupted| E[Redownload]
    B -->|Timeout| F[Increase Timeout]

    C --> G[Success]
    D --> H[Raise Exception]
    E --> G
    F --> G
    class STACFetchError(Exception):
    """Base class for STAC fetch errors."""
    pass

class AssetNotFoundError(STACFetchError):
    """Requested asset not in catalog."""
    pass

class DownloadError(STACFetchError):
    """Network/download error."""
    pass

def safe_fetch(fetcher, **kwargs) -> dict:
    """Fetch with error handling."""
    try:
        return fetcher.fetch(**kwargs)

    except AssetNotFoundError as e:
        logger.error(f"Asset not found: {e}")
        return {
            'error': 'ASSET_NOT_FOUND',
            'message': str(e),
            'suggestion': 'Check if region is in STAC catalog'
        }

    except DownloadError as e:
        logger.error(f"Download failed: {e}")
        return {
            'error': 'DOWNLOAD_FAILED',
            'message': str(e),
            'suggestion': 'Check network connection'
        }

    except Exception as e:
        logger.exception("Unexpected error")
        return {
            'error': 'UNKNOWN_ERROR',
            'message': str(e)
        }

    Async Downloads¶

    import asyncio

async def fetch_multiple():
    """Fetch multiple assets concurrently."""
    tasks = [
        fetcher.afetch(collection='lulc', year=2021),
        fetcher.afetch(collection='lulc', year=2022),
        fetcher.afetch(collection='lulc', year=2023)
    ]

    results = await asyncio.gather(*tasks)
    return results

    Pipeline Registration¶

    # computing/pipeline_registry.py

PIPELINE_STAC_DEPS = {
    'swb': {
        'required_assets': ['lulc', 'mws', 'block'],
        'optional_assets': ['dem'],
        'default_year': 2023
    },
    'hydrology': {
        'required_assets': ['dem', 'mws', 'rainfall'],
        'optional_assets': ['soils', 'lulc'],
        'default_year': 2023
    },
    'terrain': {
        'required_assets': ['dem', 'block'],
        'optional_assets': [],
        'default_year': None  # Use latest
    }
}

def get_pipeline_deps(pipeline: str) -> dict:
    """Get STAC dependencies for a pipeline."""
    return PIPELINE_STAC_DEPS.get(pipeline, {})

    Automatic Dependency Resolution¶

    def auto_fetch_for_pipeline(pipeline: str, region: dict) -> dict:
    """Automatically fetch all dependencies for a pipeline."""

    deps = get_pipeline_deps(pipeline)
    fetcher = STACFetcher()

    paths = {}

    # Fetch required assets
    for asset in deps['required_assets']:
        path = fetcher.fetch(
            collection=get_collection_for_asset(asset),
            assets=[asset],
            **region
        )
        paths[asset] = path[asset]

    # Fetch optional assets if available
    for asset in deps['optional_assets']:
        try:
            path = fetcher.fetch(
                collection=get_collection_for_asset(asset),
                assets=[asset],
                **region
            )
            paths[asset] = path[asset]
        except AssetNotFoundError:
            logger.warning(f"Optional asset {asset} not found")

    return paths

    Best Practices¶

    1. Reuse cache - Don't clear cache unnecessarily
    2. Check before fetch - Use check_cache() to avoid re-downloads
    3. Handle errors - Always wrap fetches in try-except
    4. Monitor size - Set reasonable cache limits
    5. Use versioning - Include year/version in cache keys
