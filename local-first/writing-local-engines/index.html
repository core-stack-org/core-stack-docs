    Writing Local Engines¶

    Local engines are Python modules that process geospatial data locally using STAC-provided assets. This guide shows you how to build custom local computation engines.

    Basic Engine Template¶

    # computing/my_engine/my_engine.py
from typing import Dict, Any
import geopandas as gpd
import rasterio
from rasterio.features import shapes
import numpy as np
from pathlib import Path

def run_my_engine(
    stac_paths: Dict[str, str],
    parameters: Dict[str, Any]
) -> Dict[str, Any]:
    """
    Template for local computation engine.

    Args:
        stac_paths: Dictionary of downloaded file paths
                   {'lulc': 'path/to/lulc.tif', ...}
        parameters: Request parameters
                   {'state': 'karnataka', ...}

    Returns:
        Dictionary with results
    """
    # 1. Load data
    data = load_data(stac_paths)

    # 2. Process
    results = process_data(data, parameters)

    # 3. Save outputs
    output_paths = save_outputs(results)

    # 4. Return
    return {
        'success': True,
        'output_files': output_paths,
        'statistics': results['stats']
    }

    Example: Vegetation Health Engine

    # computing/vegetation/veg_health.py

def calculate_veg_health(ndvi_path: str, threshold: float = 0.3) -> dict:
    """
    Calculate vegetation health from NDVI.

    Classifies pixels as:
    - Healthy: NDVI > 0.5
    - Moderate: 0.3 < NDVI <= 0.5
    - Poor: NDVI <= 0.3
    """
    with rasterio.open(ndvi_path) as src:
        ndvi = src.read(1)
        profile = src.profile

    # Classify
    health_classes = np.zeros_like(ndvi, dtype=np.uint8)
    health_classes[ndvi > 0.5] = 1  # Healthy
    health_classes[(ndvi > 0.3) & (ndvi <= 0.5)] = 2  # Moderate
    health_classes[ndvi <= 0.3] = 3  # Poor

    # Calculate statistics
    total_pixels = ndvi.size
    stats = {
        'healthy_pct': np.sum(health_classes == 1) / total_pixels * 100,
        'moderate_pct': np.sum(health_classes == 2) / total_pixels * 100,
        'poor_pct': np.sum(health_classes == 3) / total_pixels * 100
    }

    return {
        'classification': health_classes,
        'profile': profile,
        'statistics': stats
    }

    Step 3: Create API Endpoint¶

    # computing/api.py
from rest_framework.decorators import api_view
from rest_framework.response import Response
from .vegetation.veg_health import calculate_veg_health
from .stac_fetcher import download_stac_dependencies

@api_view(['POST'])
    def vegetation_health(request):
        """
        API endpoint for vegetation health analysis.

        Example route shape: POST /api/v1/your-new-endpoint/

    Request:
    {
        "state": "karnataka",
        "district": "raichur",
        "block": "devadurga",
        "year": 2023
    }
    """
    # Get parameters
    state = request.data.get('state')
    district = request.data.get('district')
    block = request.data.get('block')
    year = request.data.get('year', 2023)

    # Download STAC dependencies
    stac_paths = download_stac_dependencies(
        pipeline='vegetation',
        state=state,
        district=district,
        block=block,
        year=year
    )

    # Run local engine
    result = calculate_veg_health(
        ndvi_path=stac_paths['ndvi'],
        threshold=0.3
    )

    # Save output
    output_path = save_raster(
        data=result['classification'],
        profile=result['profile'],
        filename=f'veg_health_{state}_{district}_{block}_{year}.tif'
    )

    return Response({
        'success': True,
        'output_url': output_path,
        'statistics': result['statistics']
    })

    def vector_analysis_engine(mws_path: str, water_body_path: str) -> dict:
    """Example: Calculate water body density per MWS."""

    # Load data
    mws = gpd.read_file(mws_path)
    water_bodies = gpd.read_file(water_body_path)

    # Spatial join
    mws_with_water = gpd.sjoin(
        mws, 
        water_bodies[['geometry', 'area_ha']],
        how='left',
        predicate='intersects'
    )

    # Calculate statistics per MWS
    results = mws_with_water.groupby('mws_id').agg({
        'area_ha': ['count', 'sum']
    }).reset_index()

    results.columns = ['mws_id', 'water_body_count', 'total_water_area_ha']

    # Calculate density
    mws_area = mws.set_index('mws_id')['area_ha']
    results['water_density_pct'] = (
        results['total_water_area_ha'] / 
        results['mws_id'].map(mws_area) * 100
    )

    return {
        'data': results.to_dict('records'),
        'total_mws': len(mws),
        'avg_water_density': results['water_density_pct'].mean()
    }

    import rasterstats

def raster_zonal_stats_engine(
    raster_path: str,
    zones_path: str
) -> dict:
    """Example: Calculate zonal statistics."""

    # Calculate statistics for each zone
    stats = rasterstats.zonal_stats(
        zones_path,
        raster_path,
        stats=['mean', 'min', 'max', 'std'],
        geojson_out=True
    )

    # Convert to GeoDataFrame
    gdf = gpd.GeoDataFrame.from_features(stats)

    return {
        'geojson': gdf.to_json(),
        'record_count': len(gdf),
        'columns': list(gdf.columns)
    }

    Combined Vector-Raster Pattern¶

    def combined_analysis_engine(
    lulc_path: str,
    mws_path: str,
    dem_path: str
) -> dict:
    """Example: Land cover by elevation zone."""

    # Load data
    mws = gpd.read_file(mws_path)

    # Calculate elevation stats per MWS
    elev_stats = rasterstats.zonal_stats(
        mws,
        dem_path,
        stats=['mean', 'min', 'max']
    )

    # Add to MWS
    for i, stat in enumerate(elev_stats):
        mws.loc[i, 'elev_mean'] = stat['mean']
        mws.loc[i, 'elev_range'] = stat['max'] - stat['min']

    # Classify elevation zones
    mws['elev_zone'] = pd.cut(
        mws['elev_mean'],
        bins=[0, 500, 1000, 1500, 3000],
        labels=['Low', 'Moderate', 'High', 'Very High']
    )

    # Calculate LULC distribution by elevation zone
    lulc_by_elev = rasterstats.zonal_stats(
        mws,
        lulc_path,
        categorical=True,
        category_map={1: 'Ag', 2: 'Forest', 3: 'Water'}
    )

    return {
        'elevation_summary': mws.groupby('elev_zone').size().to_dict(),
        'lulc_by_elevation': lulc_by_elev
    }

    Lazy Loading¶

    from contextlib import contextmanager

@contextmanager
def load_raster(path: str):
    """Context manager for safe raster loading."""
    src = rasterio.open(path)
    try:
        yield src
    finally:
        src.close()

# Usage
with load_raster('dem.tif') as src:
    data = src.read(1)
    profile = src.profile

    Caching Results¶

    from functools import lru_cache
import hashlib

def get_cache_key(*args) -> str:
    """Generate cache key from arguments."""
    key_str = '|'.join(str(a) for a in args)
    return hashlib.md5(key_str.encode()).hexdigest()

def cached_analysis(stac_paths: dict, params: dict) -> dict:
    """Cache analysis results to disk."""
    cache_key = get_cache_key(stac_paths, params)
    cache_path = Path(f'cache/{cache_key}.json')

    if cache_path.exists():
        return json.loads(cache_path.read_text())

    # Run analysis
    result = run_analysis(stac_paths, params)

    # Cache result
    cache_path.parent.mkdir(exist_ok=True)
    cache_path.write_text(json.dumps(result))

    return result

    Error Handling¶

    from typing import Union
import logging

logger = logging.getLogger(__name__)

def safe_engine_wrapper(
    engine_func,
    stac_paths: dict,
    params: dict
) -> Union[dict, dict]:
    """
    Wrapper for safe engine execution.

    Returns:
        Success: {'success': True, 'data': ...}
        Error: {'success': False, 'error': ...}
    """
    try:
        # Validate inputs
        if not stac_paths:
            raise ValueError("No STAC paths provided")

        # Check files exist
        for key, path in stac_paths.items():
            if not Path(path).exists():
                raise FileNotFoundError(f"Missing {key}: {path}")

        # Run engine
        result = engine_func(stac_paths, params)

        return {
            'success': True,
            'data': result
        }

    except FileNotFoundError as e:
        logger.error(f"Data file not found: {e}")
        return {
            'success': False,
            'error': 'DATA_NOT_FOUND',
            'message': str(e)
        }

    except MemoryError as e:
        logger.error(f"Memory error: {e}")
        return {
            'success': False,
            'error': 'MEMORY_ERROR',
            'message': 'Region too large for local processing'
        }

    except Exception as e:
        logger.exception("Engine error")
        return {
            'success': False,
            'error': 'PROCESSING_ERROR',
            'message': str(e)
        }

    Testing Your Engine¶

    # tests/test_vegetation_engine.py
import pytest
from computing.vegetation.veg_health import calculate_veg_health
import numpy as np
import rasterio
from rasterio.io import MemoryFile

def create_test_ndvi(data: np.ndarray):
    """Create in-memory test raster."""
    profile = {
        'driver': 'GTiff',
        'height': data.shape[0],
        'width': data.shape[1],
        'count': 1,
        'dtype': data.dtype,
        'crs': 'EPSG:4326',
        'transform': rasterio.Affine.identity()
    }

    with MemoryFile() as memfile:
        with memfile.open(**profile) as src:
            src.write(data, 1)
        return memfile

class TestVegetationHealth:
    def test_all_healthy(self):
        """Test with all healthy vegetation."""
        ndvi = np.ones((10, 10), dtype=np.float32) * 0.8

        with create_test_ndvi(ndvi) as memfile:
            result = calculate_veg_health(memfile.name)

        assert result['statistics']['healthy_pct'] == 100
        assert result['statistics']['poor_pct'] == 0

    def test_mixed_vegetation(self):
        """Test with mixed vegetation health."""
        ndvi = np.array([
            [0.6, 0.6, 0.4, 0.4],
            [0.6, 0.6, 0.4, 0.4],
            [0.2, 0.2, 0.1, 0.1],
            [0.2, 0.2, 0.1, 0.1]
        ], dtype=np.float32)

        with create_test_ndvi(ndvi) as memfile:
            result = calculate_veg_health(memfile.name)

        assert result['statistics']['healthy_pct'] == 25
        assert result['statistics']['moderate_pct'] == 25
        assert result['statistics']['poor_pct'] == 50

    Chunked Processing¶

    def process_large_raster(raster_path: str, chunk_size: int = 1024):
    """Process raster in chunks to manage memory."""

    with rasterio.open(raster_path) as src:
        height, width = src.height, src.width

        for y in range(0, height, chunk_size):
            for x in range(0, width, chunk_size):
                # Read chunk
                window = rasterio.windows.Window(
                    x, y,
                    min(chunk_size, width - x),
                    min(chunk_size, height - y)
                )
                chunk = src.read(1, window=window)

                # Process chunk
                result = process_chunk(chunk)

                # Write result
                yield result

    Parallel Processing¶

    from multiprocessing import Pool
import geopandas as gpd

def parallel_vector_analysis(
    gdf: gpd.GeoDataFrame,
    analysis_func,
    n_workers: int = 4
) -> list:
    """Process GeoDataFrame in parallel."""

    # Split into chunks
    chunks = np.array_split(gdf, n_workers)

    # Process in parallel
    with Pool(n_workers) as pool:
        results = pool.map(analysis_func, chunks)

    # Combine results
    return pd.concat(results)
